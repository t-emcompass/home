OPENSCAD CODE


////////////////////////////////////////////////////////////
// SPLIT CORE FERRITE HOUSING 
// - correct hinge alignment
// - real cable feedthrough
// - true full-moon split
////////////////////////////////////////////////////////////

$fn = 120;

// ---------------- CORE ----------------
core_od = 26.9;
core_id = 16.4;
core_h  = 30;

clear = 0.4;
wall  = 2.5;

cav_od = core_od + 2*clear;
cav_id = core_id - 2*clear;

height = core_h + (2*clear)+5;

// outer
outer_od = cav_od + 2*wall;

// ---------------- HINGE ----------------
hinge_pin_d = 3.2;
hinge_lug_w = 6;
hinge_lug_h = 6;

// ---------------- CABLE ----------------
wire_d = 1.0;
wire_clear = 0.5;
wire_hole_d = wire_d + wire_clear;

// ---------------- MODE ----------------
// 0 = assembly
// 1 = half A
// 2 = half B
part = 0;


sma_x = outer_od/2;
sma_y = 0;
sma_z = height/2;


// ============================================================
// FULL RING (base geometry)
// ============================================================
module full_ring()
{
    difference()
    {
        // outer
        cylinder(h=height, r=outer_od/2);

        // ferrite cavity (true torus cut)
        translate([0,0,2.5])
            cylinder(h=height-5, r=cav_od/2);

        translate([0,0,-1])
            cylinder(h=height+2, r=cav_id/2);

        // central cable feedthrough 
        translate([0,0,0])
            cylinder(h=height+2, r=wire_hole_d/2);
    }
}


// ============================================================
// HALF CUT (true moon split)
// ============================================================
module half(body=true)
{
    intersection()
    {
        full_ring();

        // cut plane (exact moon split)
        translate([-outer_od, -outer_od/2, -1])
            cube([outer_od, outer_od, height+2]);
    }
}


// ============================================================
// HINGE LUGS (aligned to split plane)
// ============================================================


module rounded_lug_2d(w, h, r)
{
    hull()
    {
        translate([-w/2 + r, -h/3])
            circle(r=r);

        translate([-w/2 + r, h/2])
            circle(r=r);

        translate([w/2, -h/2])
            square([0.01, h]);
    }
}




module hinge_lugs(side=1)
{
    lug_w = hinge_lug_w+3;
    lug_h = hinge_lug_h+2 ;
    fillet_r = 2;

    translate([-1.7, side*((outer_od/2)+4.8), height/4])
    difference()
    {
        union() {
    // Block1
    translate([0, 0, 0])
        linear_extrude(height = height/4, center=false)
            rounded_lug_2d(lug_w, lug_h, fillet_r);

    // block 2)
    translate([0, 0, height/2])
        linear_extrude(height = height/4-2, center=false)
            rounded_lug_2d(lug_w, lug_h, fillet_r);
}

        // hinge hole
        translate([lug_w/3 - fillet_r*0.6, -1, 0])
            rotate([180,0,0])
                cylinder(h=lug_h*10, r=hinge_pin_d/2, center=true);
    }
}


// ============================================================
// SMA HOLE 
// ============================================================
module sma_hole()
{
    translate([sma_x+15, sma_y, sma_z])
        rotate([0,90,0])
            cylinder(h=outer_od*2, r=6.5/2, center=true);
}

module sma_support_block()
{
    translate([
        outer_od*2-11,   
        0,
        height/2
    ])
    cube([6, 13, 13], center=true);
}


// ============================================================
// HALF SHELL A
// ============================================================
module half_A()
{
    difference()
    {
        union()
        {
            half();
            sma_support_block();
            //sma_hole();
        }

        sma_hole();
    }

     mirror([0,1,0])hinge_lugs(-1);
}


// ============================================================
// HALF SHELL B
// ============================================================
module half_B()
{
    difference()
    {
        rotate([0,0,180]) half();
        sma_hole();
       
    }
    
   mirror([1,0,0])hinge_lugs(-1);
}


// ============================================================
// ASSEMBLY VIEW
// ============================================================
module assembly()
{
    color("silver") half_A();

    translate([outer_od + 5, 0, 0])
        color("gainsboro") half_B();
}


// ============================================================
// OUTPUT
// ============================================================
if (part == 0) assembly();
if (part == 1) half_A();
if (part == 2) half_B();
